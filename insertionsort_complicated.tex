% define a function that extracts the value for which will be sorted from your
% data structure
\long\def\inso@C@getvalue#1%>>=
  {%
    \inso@ifdimen{#1}
      {\def\inso@C@value{#1}}
      {\def\inso@C@value{#1\inso@unit}}%
  }%=<<
\long\def\inso@C@ifsmaller#1#2%>>=
  {%
    \ifdim#1<#2\relax
      \inso@fi@firstoftwo
    \fi
    \inso@secondoftwo
  }%=<<
\long\def\inso@C@insert@last#1\fi#2\inso@qstop#3%>>=
  {%
    \fi
    \edef\inso@C@sorted{\unexpanded\expandafter{\inso@C@sorted{#1}{#3}}}%
  }%=<<
\long\def\inso@C@insert@place#1#2\inso@qmark\inso@qmark\inso@qstop#3%>>=
  {%
    \edef\inso@C@sorted{\unexpanded\expandafter{\inso@C@tmp{#1}{#3}#2}}%
  }%=<<
\long\def\inso@C@insert@b#1#2#3%>>=
  {%
    % #1: new value
    % #2: next value
    % #3: next element
    \ifx\inso@qmark#3%
      \inso@C@insert@last{#1}%
    \fi
    \inso@C@ifsmaller{#1}{#2}%
      {\inso@C@insert@place{#1}{#2}{#3}}%
      {%
        \edef\inso@C@tmp{\unexpanded\expandafter{\inso@C@tmp{#2}{#3}}}%
        \inso@C@insert@b{#1}%
      }%
  }%=<<
\long\def\inso@C@insert@a#1#2%>>=
  {%
    \def\inso@C@tmp{}%
    \inso@C@insert@b{#2}#1\inso@qmark\inso@qmark\inso@qstop
  }%=<<
\long\def\inso@C@insert#1#2%>>=
  {%
    % #1: value in dimen
    % #2: full element
    \expandafter\inso@C@insert@a\expandafter{\inso@C@sorted}{#1}{#2}%
  }%=<<
\long\def\inso@C@sort#1,%>>=
  {%
    \inso@ifblank{#1}%
      {\inso@blankelementerror}%
      {%
        \ifx\inso@qmark#1%
          \inso@fi@gobbletoqstop
        \fi
        \inso@C@getvalue{#1}%
        \expandafter\inso@C@insert\expandafter{\inso@C@value}{#1}%
      }%
    \inso@C@sort
  }%=<<
\long\def\inso@C@startsort#1,%>>=
  {%
    \inso@C@getvalue{#1}%
    \edef\inso@C@sorted
      {{\unexpanded\expandafter{\inso@C@value}}{\unexpanded{#1}}}%
    \inso@C@sort
  }%=<<
\long\def\inso@C@clistize@#1#2%>>=
  {%
    \ifx\inso@qmark#1%
      \inso@fi@gobble
    \fi
    \inso@firstofone
    {%
      \unexpanded{,{#2}}%
      \inso@C@clistize@
    }%
  }%=<<
\long\def\inso@C@clistize#1#2%>>=
  {%
    \unexpanded{{#2}}%
    \inso@C@clistize@
  }%=<<
\protected\long\def\inso@C@insertionsort#1%>>=
  {%
    \inso@C@startsort#1,\inso@qmark,\inso@qstop
    \edef\inso@C@sorted
      {\expandafter\inso@C@clistize\inso@C@sorted\inso@qmark\inso@qmark}%
  }%=<<
\protected\long\def\insertionsortC#1#2%>>=
  {%
    \inso@ifblank{#2}%
      {\def#1{}}%
      {%
        \begingroup
          \inso@C@insertionsort{#2}%
        \expandafter
        \endgroup
        \expandafter\def\expandafter#1\expandafter{\inso@C@sorted}%
      }%
  }%=<<
\protected\long\def\InsertionsortC#1#2#3%>>=
  {%
    \inso@ifblank{#3}%
      {\def#2{}}%
      {%
        \begingroup
          \long\def\inso@C@getvalue##1{#1}%
          \inso@C@insertionsort{#3}%
        \expandafter
        \endgroup
        \expandafter\def\expandafter#2\expandafter{\inso@C@sorted}%
      }%
  }%=<<
