\input "insertionsort_shared.tex"

\long\def\inso@S@ifsmaller@#1#2%>>=
  {%
    \ifdim#1<#2\relax
      \inso@fi@secondofthree
    \fi
    \inso@secondoftwo
  }%=<<
\protected\long\def\inso@S@ifsmaller#1#2%>>=
  {%
    \inso@ifdimen{#2}
      {%
        \inso@ifdimen{#1}
          {\inso@S@ifsmaller@{#1}}
          {\inso@S@ifsmaller@{#1pt}}%
        {#2}%
      }
      {%
        \inso@ifdimen{#1}
          {\inso@S@ifsmaller@{#1}}
          {\inso@S@ifsmaller@{#1pt}}%
        {#2pt}%
      }%
  }%=<<
\long\def\inso@S@insert@last#1\fi#2\inso@qmark#3\inso@qstop%>>=
  {%
    \fi
    \edef\inso@S@sorted{\unexpanded\expandafter{\inso@S@sorted{#1}}}%
  }%=<<
\long\def\inso@S@insert@place#1#2\inso@qmark#3\inso@qstop%>>=
  {%
    \edef\inso@S@sorted
      {%
        \unexpanded\expandafter{\inso@S@tmp{#1}}%
        \unexpanded\expandafter{\inso@gobble#2}%
      }%
  }%=<<
\protected\long\def\inso@S@insert@a#1#2%>>=
  {%
    % #1: new value
    % #2: next value
    \ifx\inso@qmark#2%
      \inso@S@insert@last{#1}%
    \fi
    \inso@S@ifsmaller{#1}{#2}%
      {\inso@S@insert@place{#1}.{#2}}%
      {%
        \edef\inso@S@tmp{\unexpanded\expandafter{\inso@S@tmp{#2}}}%
        \inso@S@insert@a{#1}%
      }%
  }%=<<
\protected\long\def\inso@S@insert#1#2%>>=
  {%
    \def\inso@S@tmp{}%
    \inso@S@insert@a{#2}#1\inso@qmark\inso@qmark\inso@qmark\inso@qstop
  }%=<<
\long\def\inso@S@sort#1,%>>=
  {%
    \inso@ifblank{#1}%
      {\inso@blankelementerror}%
      {%
        \ifx\inso@qmark#1%
          \inso@fi@gobbleuntilqstop
        \fi
        \expandafter\inso@S@insert\expandafter{\inso@S@sorted}{#1}%
      }%
    \inso@S@sort
  }%=<<
\long\def\inso@S@clistize@#1%>>=
  {%
    \ifx\inso@qmark#1%
      \inso@fi@gobbletwo
    \fi
    \inso@firstofone
    {%
      \unexpanded{,{#1}}%
      \inso@S@clistize@
    }%
  }%=<<
\long\def\inso@S@clistize#1%>>=
  {%
    \unexpanded{{#1}}%
    \inso@S@clistize@
  }%=<<
\protected\long\def\inso@S@insertionsort#1%>>=
  {%
    \edef\inso@S@sorted{{\inso@getfirst#1,\inso@qstop}}%
    \expandafter\inso@S@sort
      \romannumeral-`0\inso@gobblefirst#1,\inso@qnil,\inso@qmark,\inso@qstop
    \edef\inso@S@sorted{\expandafter\inso@S@clistize\inso@S@sorted\inso@qmark}%
  }%=<<
\protected\long\def\insertionsortS#1#2%>>=
  {%
    \inso@ifblank{#2}%
      {\def#1{}}%
      {%
        \begingroup
          \inso@S@insertionsort{#2}%
        \expandafter
        \endgroup
        \expandafter\def\expandafter#1\expandafter{\inso@S@sorted}%
      }%
  }%=<<
\protected\long\def\InsertionsortS#1#2#3%>>=
  {%
    \inso@ifblank{#3}%
      {\def#2{}}%
      {%
        \begingroup
          \long\def\inso@S@ifsmaller##1##2{#1}%
          \inso@S@insertionsort{#3}%
        \expandafter
        \endgroup
        \expandafter\def\expandafter#2\expandafter{\inso@S@sorted}%
      }%
  }%=<<
